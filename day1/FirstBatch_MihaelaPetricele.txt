
CREATE TABLE country(id varchar(255) PRIMARY KEY DEFAULT 'XX')
CREATE TABLE country_details(id int PRIMARY KEY, population int, climate-type varchar(255), country_id varchar(255), FOREIGN KEY(country_id) REFERENCES country(id))
CREATE TABLE city (id int PRIMARY KEY, name varchar(255) UNIQUE, country_id varchar(255), FOREIGN KEY (country_id) REFERENCES country(id)) 
CREATE TABLE resident (id int PRIMARY KEY, name varchar(255), birth_date DATE ) 
CREATE TABLE city_resident(id int PRIMARY KEY, city_id int, resident_id int, FOREIGN KEY (city_id) REFERENCES city(id), FOREIGN KEY (resident_id) REFERENCES resident(id))
